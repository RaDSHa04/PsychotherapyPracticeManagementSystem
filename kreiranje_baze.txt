CREATE TABLE cena_seanse (
    cena_id int  NOT NULL,
    cena int  NOT NULL,
    datumPromene date  NOT NULL,
    CONSTRAINT cena_seanse_pk PRIMARY KEY (cena_id)
);

CREATE INDEX idx_cena_datum
ON cena_seanse (cena, datumPromene);

ALTER TABLE cena_seanse
ADD CONSTRAINT unique_cena_datum
UNIQUE (cena, datumPromene);

-- Table: centar_za_obuku
CREATE TABLE centar_za_obuku (
    centar_id int  NOT NULL,
    nazivCentra nvarchar(50)  NOT NULL,
    mejl nvarchar(50)  NOT NULL,
    brojTelefona varchar(20)  NOT NULL,
    postanskaAdresa nvarchar(100)  NOT NULL,
    CONSTRAINT centar_za_obuku_pk PRIMARY KEY (centar_id)
);

ALTER TABLE centar_za_obuku
ADD CONSTRAINT unique_naziv_adresa
UNIQUE (nazivCentra, postanskaAdresa);

CREATE INDEX idx_naziv_mejl
ON centar_za_obuku (nazivCentra, mejl);

-- Table: drziSeansu
CREATE TABLE drziSeansu (
    drzi_id int  NOT NULL,
    psihoterapeut_psihoterapeut_id int,
    kandidat_kandidat_id int,
    kandidat_centar_za_obuku_centar_id int,
    CONSTRAINT drziSeansu_pk PRIMARY KEY (drzi_id)
);


-- Table: fakultet
CREATE TABLE fakultet (
    fakultet_id int  NOT NULL,
    nazivFakulteta nvarchar(50)  NOT NULL,
    univerzitet_univerzitet_id int  NOT NULL,
    CONSTRAINT fakultet_pk PRIMARY KEY (fakultet_id,univerzitet_univerzitet_id)
);

ALTER TABLE fakultet
ADD CONSTRAINT unique_naziv_univerzitet
UNIQUE (nazivFakulteta, univerzitet_univerzitet_id);

CREATE INDEX idx_naziv_fakulteta
ON fakultet (nazivFakulteta);

-- Table: kandidat
CREATE TABLE kandidat (
    kandidat_id int  NOT NULL,
    ime nvarchar(50)  NOT NULL,
    prezime nvarchar(50)  NOT NULL,
    jmbg char(13)  NOT NULL,
    datumRodjenja date  NOT NULL,
    prebivaliste nvarchar(50)  NOT NULL,
    brojTelefona varchar(20)  NOT NULL,
    mejl nvarchar(50)  NOT NULL,
    fakultet_fakultet_id int  NOT NULL,
    fakultet_univerzitet_univerzitet_id int  NOT NULL,
    stepen_studija_stepen_id int  NOT NULL,
    centar_za_obuku_centar_id int  NOT NULL,
    CONSTRAINT kandidat_pk PRIMARY KEY (kandidat_id,centar_za_obuku_centar_id)
);

ALTER TABLE kandidat
ADD CONSTRAINT unique_jmbg
UNIQUE (jmbg);

CREATE INDEX idx_ime_prezime_mejl
ON kandidat (ime, prezime, mejl);


-- Table: klijent
CREATE TABLE klijent (
    klijent_id int  NOT NULL,
    ime nvarchar(50)  NOT NULL,
    prezime nvarchar(50)  NOT NULL,
    datumRodjenja date  NOT NULL,
    pol nvarchar(50)  NOT NULL,
    mejl nvarchar(50)  NOT NULL,
    brojTelefona varchar(20)  NOT NULL,
    isaoRanije boolean  NOT NULL,
    opisProblema nvarchar(500)  NOT NULL,
    CONSTRAINT klijent_pk PRIMARY KEY (klijent_id)
);

ALTER TABLE klijent
ADD CONSTRAINT unique_ime_prezime_mejl
UNIQUE (ime, prezime, mejl);

CREATE INDEX idx_ime_prezime_mejl
ON klijent (ime, prezime, mejl);

-- Table: oblast
CREATE TABLE oblast (

ALTER TABLE oblast

-- Table: placanje

    seansa_seansa_id int  NOT NULL,
    seansa_klijent_klijent_id int  NOT NULL,
    seansa_drziSeansu_drzi_id int  NOT NULL,

ALTER TABLE placanje

ALTER TABLE placanje DROP PRIMARY KEY;

ALTER TABLE placanje
ADD CONSTRAINT placanje_pk
PRIMARY KEY (placanje_id, seansa_drziSeansu_drzi_id, seansa_klijent_klijent_id, seansa_seansa_id);
